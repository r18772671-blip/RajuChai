CREATE TABLE Student(
  rollno INT PRIMARY KEY,
  name   VARCHAR(50),
  class  VARCHAR(10)
);

CREATE TABLE Marks(
  rollno  INT,
  subject VARCHAR(50),
  marks   INT
);

SELECT * FROM Student;

SELECT * FROM Marks;

SELECT s.name, m.subject, m.marks
FROM Student s
INNER JOIN Marks m ON s.rollno = m.rollno;

SELECT s.name, m.subject, m.marks
FROM Student s
LEFT JOIN Marks m ON s.rollno = m.rollno
ORDER BY s.rollno, m.subject;

SELECT s.name, m.subject, m.marks
FROM Student s
RIGHT JOIN Marks m ON s.rollno = m.rollno
ORDER BY m.rollno, m.subject;

SELECT s.name, m.subject, m.marks
FROM Student s
JOIN Marks m ON s.rollno = m.rollno
WHERE m.marks > 70;

SELECT name
FROM Student
WHERE rollno IN (SELECT rollno FROM Marks);

SELECT name
FROM Student
WHERE rollno NOT IN (SELECT rollno FROM Marks);

SELECT rollno, COUNT(*) AS total_subjects
FROM Marks
GROUP BY rollno
ORDER BY rollno;

CREATE VIEW student_report AS
SELECT s.rollno, s.name, s.class, m.subject, m.marks
FROM Student s
JOIN Marks m ON s.rollno = m.rollno;

SELECT * FROM student_report ORDER BY rollno, subject;
